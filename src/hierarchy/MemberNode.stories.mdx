import { Canvas, Meta, Story } from "@storybook/addon-docs";
import { toReact } from "../d3-component/toReact";

import { MemberNode } from "./MemberNode";

export const ReactMemberNode = toReact(MemberNode, "svg");
export const argTypes = {};

<Meta title="MemberNode" component={ReactMemberNode} argTypes={argTypes} />

export const Template = (args) => {
  return (<ReactMemberNode {...args} />);
};

# Minimal

A name, a size, and two colors are enough to render a basic member node. Hover the node to see the name of the member.

<Canvas>
  <Story name="Minimal" args={{
    type: "member",
    color: "#F6B825",
    backgroundColor: "#122686",
    size: 20,
    name: "Vincent Van Gogh",
  }}>
    {Template.bind({})}
  </Story>
</Canvas>

# Complete

But a picture of the member, date of birth, and date of death can also be provided. Hover the node to see both the name and birth/death dates of the member.

<Canvas>
  <Story name="Complete" args={{
    type: "member",
    color: "#F6B825",
    backgroundColor: "#122686",
    size: 20,
    name: "Vincent Van Gogh",
    picture: "https://upload.wikimedia.org/wikipedia/commons/b/b2/Vincent_van_Gogh_-_Self-Portrait_-_Google_Art_Project.jpg",
    birth: [1853, 3, 30],
    death: [1890, 7, 29],
  }}>
    {Template.bind({})}
  </Story>
</Canvas>
